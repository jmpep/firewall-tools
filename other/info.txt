import requests
import json

class CheckpointAPI:

    def __init__(self, server, user, password):
        self.server = server
        self.base = f"https://{server}/web_api"
        self.sid = None

        payload = {
            "user": user,
            "password": password
        }

        r = requests.post(
            f"{self.base}/login",
            json=payload,
            verify=False
        )

        r.raise_for_status()

        self.sid = r.json()["sid"]

        self.headers = {
            "X-chkp-sid": self.sid
        }

    def api_call(self, command, payload=None):

        if payload is None:
            payload = {}

        r = requests.post(
            f"{self.base}/{command}",
            json=payload,
            headers=self.headers,
            verify=False
        )

        r.raise_for_status()

        return r.json()

    def logout(self):
        self.api_call("logout")

import requests

urllib3.disable_warnings()

class CheckPointAPI:

    def __init__(self, server, user, password):

        self.base = f"https://{server}/web_api"

        r = requests.post(
            f"{self.base}/login",
            json={
                "user": user,
                "password": password
            },
            verify=False,
            timeout=60
        )

        r.raise_for_status()

        self.sid = r.json()["sid"]

        self.headers = {
            "X-chkp-sid": self.sid
        }

    def call(self, command, payload=None):

        if payload is None:
            payload = {}

        r = requests.post(
            f"{self.base}/{command}",
            headers=self.headers,
            json=payload,
            verify=False,
            timeout=300
        )

        r.raise_for_status()

        return r.json()

    def logout(self):
        self.call("logout")

def paged_api(api, command, payload, key):

    offset = 0
    limit = 500

    results = []

    while True:

        p = payload.copy()

        p["offset"] = offset
        p["limit"] = limit

        data = api.call(command, p)

        chunk = data.get(key, [])

        results.extend(chunk)

        if len(chunk) < limit:
            break

        offset += limit

    return results

Collect all UIDs recursively

This is the critical part.

def collect_uids(node, uid_set):

    if isinstance(node, dict):

        uid = node.get("uid")

        if uid:
            uid_set.add(uid)

        for value in node.values():
            collect_uids(value, uid_set)

    elif isinstance(node, list):

        for item in node:
            collect_uids(item, uid_set)

For large environments.

from concurrent.futures import ThreadPoolExecutor
def fetch_object(api, uid):

    try:

        obj = api.call(
            "show-object",
            {
                "uid": uid,
                "details-level": "full"
            }
        )

        return uid, obj

    except Exception as e:

        return uid, {
            "error": str(e)
        }
def download_objects(api, uid_set):

    objects = {}

    with ThreadPoolExecutor(
        max_workers=20
    ) as pool:

        futures = []

        for uid in uid_set:

            futures.append(
                pool.submit(
                    fetch_object,
                    api,
                    uid
                )
            )

        for f in futures:

            uid, obj = f.result()

            objects[uid] = obj

    return objects
Verify consistency
def verify_uids(
    uid_set,
    object_db
):

    missing = []

    for uid in uid_set:

        if uid not in object_db:

            missing.append(uid)

    return missing
Export policy + objects
def export_policy_objects(
    api,
    package_name
):

    package = export_package(
        api,
        package_name
    )

    uids = set()

    collect_uids(
        package,
        uids
    )

    objects = download_objects(
        api,
        uids
    )

    missing = verify_uids(
        uids,
        objects
    )

    return {

        "package": package,

        "objects": objects,

        "missing_uids": missing
    }
